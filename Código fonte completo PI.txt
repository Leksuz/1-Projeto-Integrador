                            if (busca > 0){
                                validaCPF = 1;
                            }else{
                                System.out.println("CPF não encontrado e/ou incorreto.");
                            }
                        }while(validaCPF==0);
                    }
                }
            }else{
                System.out.println("Funcionalidade não identificada.\nPor favor digite novamente.");
            }
        }while (entrada!=0);//Fim do programa.
    }
